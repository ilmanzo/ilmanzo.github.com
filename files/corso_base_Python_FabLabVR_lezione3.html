def inversa(s):
    """Ritorna la stringa inversa della stringa ricevuta in input."""
    s_inversa = ""
    n = len(s)
    for i in range(n, 0, -1): 
        s_inversa += s[i - 1] 
    return s_inversa

def is_palindroma(s):
    """Ritorna True se la stringa specificata è palindroma"""
    return s.lower() == inversa(s).lower()

print(is_palindroma("Anna"))

n_da_indovinare = random.randint(1, 100)

while not indovinato and tentativi <= 3:

    n = input("Tentativo "+ str(tentativi) +", Numero: ")

    # Verifica se ha indovinato oppure non è così.

        indovinato = True

        print("Ho pensato un numero più grande\n")

    # Prossimo tentativo
    tentativi += 1

fattoriale ricorsivo (2)

n! = n(n-1)(n-2)(n-3) ... 2*1

quindi 4! = 432*1 = 24

quindi 5! = 5(4321) = 120

ovvero 5! = 5*4!

n! = n*(n-1)!

... ma 0! = 1

def fattoriale_r(n):
  if n == 0:
      return 1
  else:
      return n * fattoriale_r(n - 1)

fibonacci iterativo e ricorsivo

def fibonacci_i(n):
    a, b = 1, 1

    for i in range(1, n + 1):
        a, b = b, a + b

    return a

def fibonacci_r(n):
    if n in (0, 1):
        return 1
    else:
        return fibonacci_r(n - 1) + fibonacci_r(n - 2)

if __name__ == '__main__':
    n = eval(input("Numero: "))

    for i in range(0, n):
        print(fibonacci_i(i), fibonacci_r(i))

Esempio di una lista alla matriciana

def count(matrice, n):
    '''Ritorna il numero di volte che n appare nella matrice
    '''
    n_volte = 0
    for row in matrice:
        for element in row:
            if element == n:
                n_volte += 1
    return n_volte

matrice = ((1, 0, 4), (3, 4, 1), (4, 5, 9))

print(count(matrice, 4))

frequenza delle vocali

def frequenza_vocali(filename):
    ''' Ritorna un dizionario con la frequenza delle
        vocali presenti in un dato file di testo.
    '''
    f = open(filename, "r")

    frequenza_vocali_dict = {"a":0, "e":0, "i":0, "o":0, "u":0}
    vocali = frequenza_vocali_dict.keys()

    for line in f.readlines():
        for carattere in line:
            if carattere in vocali:
                frequenza_vocali_dict[carattere] += 1

    f.close()
    return frequenza_vocali_dict
