iteratori e iterabili

class ListaCircolare(object):
  def __init__(self):
    self.lista=[] # lista interna
    self.indice=0
  def append(self,dato):
    self.lista.append(dato)
  def __index__(self):
    return self.lista[self.indice]
  def __iter__(self):
    return self
  def __next__(self):
    self.indice+=1
    if self.indice>=len(self.lista):
      self.indice=0
    return self.lista[self.indice]

lst=ListaCircolare()
lst.append(5)
lst.append(8)
lst.append(42)
lst.append(77)
for elem in lst:
  print(elem)

thread e code

molto spesso negli algoritmi paralleli si tratta di preparare una "coda" di lavori e un certo numero di "lavoratori" che prendono un dato dalla coda, lo elaborano e lo restituiscono (FIFO) Python offre nella libreria standard una classe Queue il cui accesso concorrente e' sincronizzato e gestito internamente in modo trasparente.

from queue import Queue
from threading import Thread
import random
import time

def do_stuff(q,id):
  while True:
    n=q.get()
    r=random.randint(5,20)
    print("Thread {0}: ricevuto {1}; dormo per {2} secondi...".format(id,n,r))
    time.sleep(r)
    q.task_done()

q = Queue(maxsize=0)
num_threads = 10

for i in range(num_threads):
  worker = Thread(target=do_stuff, args=(q,i))
  worker.setDaemon(True)
  worker.start()

for x in range(100):
  q.put(x)
  time.sleep(1)

q.join() #aspetta che la coda sia vuota

from tkinter import *

class MyApp:
  def __init__(self, parent):
    self.myParent = parent
    self.myContainer1 = Frame(parent)
    self.myContainer1.pack()
    self.button1 = Button(self.myContainer1, command=self.button1Click)
    self.button1.configure(text="OK", background= "green")
    self.button1.pack(side=LEFT)
    self.button1.focus_force()
    self.button2 = Button(self.myContainer1, command=self.button2Click)
    self.button2.configure(text="Cancel", background="red")
    self.button2.pack(side=RIGHT)

  def button1Click(self):  ### (3)
    print("button1Click event handler")
    if self.button1["background"] == "green":
      self.button1["background"] = "yellow"
    else:
      self.button1["background"] = "green"

  def button2Click(self): ### (4)
    print("button2Click event handler")
    self.myParent.destroy()

root = Tk()
myapp = MyApp(root)
root.mainloop()

esempio pyside/qt

from PySide import QtGui, QtUiTools
import sys

class MyWindow(QtGui.QMainWindow):
  def __init__(self, *args):  
    QtGui.QMainWindow.__init__(self, *args)
    self.ui = QtUiTools.QUiLoader().load("esempioqt.ui")
    self.connetti_eventi()
    self.testo=[] #uso per memorizzare le righe

  def show(self):
    self.ui.show()
    self.ui.raise_()

  def connetti_eventi(self):
    self.ui.struca.clicked.connect(self.aggiungiriga)
    self.ui.scancela.clicked.connect(self.azzeratesto)

  def aggiungiriga(self):
    self.testo.append(self.ui.riga_inserimento.text())
    self.ui.casellatesto.setText("\n".join(self.testo))

  def azzeratesto(self):
    self.testo.clear()
    self.ui.casellatesto.setText("")

app = QtGui.QApplication(sys.argv)
window=MyWindow()
window.show()
sys.exit(app.exec_())

  Serial.begin(9600);
  pinMode(ledPin, OUTPUT);
}

void loop(){

  if (Serial.available() > 0) {
    // legge il byte
    incomingByte = Serial.read();
    if(incomingByte == 49) { // ASCII 49=='1'
      digitalWrite(ledPin, HIGH);
      delay(500); //ritardo per rendere apprezzabile
    }

      digitalWrite(ledPin, LOW);

  }

socket di rete: client

client.py

import socket
import sys

if len(sys.argv)<2:
    print("uso: client.py stringa")
    sys.exit(1)

HOST, PORT = "localhost", 9999
data = " ".join(sys.argv[1:])

# Crea un socket (SOCK_STREAM = TCP , SOCK_DGRAM = UDP)
sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

try:
    # Si connette al server e spedisce i dati
    sock.connect((HOST, PORT))
    sock.sendall(bytes(data + "\n", "utf-8"))
    # Ricevi dati e chiude comunicazione
    received = str(sock.recv(1024), "utf-8")
finally:
    sock.close()

print("Spedito:  {}".format(data))
print("Ricevuto: {}".format(received))

socket di rete: server

server.py

dovrei fare listen(), accept(), eventualmente lanciare un thread separato per gestire N connessioni ... Per semplificarmi la vita posso usare il modulo di libreria socketserver

import socketserver

class MyServer(socketserver.BaseRequestHandler):

    def handle(self):
        self.data = self.request.recv(1024).strip()
        print("ricevo da {}".format(self.client_address[0]))
        print(self.data)
        # just send back the same data, but upper-cased
        self.request.sendall(self.data.upper())

HOST, PORT = "0.0.0.0", 9999
# Crea il server, facendo il bind alla porta 9999
server = socketserver.TCPServer((HOST, PORT), MyServer)
# Attiva il server fino a che non premo CTRL-C 
print("Server in esecuzione sulla porta {}".format(PORT))
server.serve_forever()

implementiamo un "negozio" di biciclette virtuale: l'utente visita il nostro sito e puo' comporre la sua bici scegliendo vari tipi di componenti.

articoli disponibili: - Telaio - Ruote - Cambio - Manubrio - Pedali

ogni articolo avra'

  • descrizione
  • prezzo
  • alcune caratteristiche proprie (es. ruote = diametro ; cambio = n.di rapporti ; manubrio = dritto/curvo ; telaio = alluminio/carbonio)

mentre sta assemblando la propria bici, l'utente visualizza l'elenco dei componenti, il prezzo totale e puo' decidere di ordinare altri pezzi di scorta
